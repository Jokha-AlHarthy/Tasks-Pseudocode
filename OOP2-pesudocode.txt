CLASS ATTEMPT
	ATTRIBUTES
		id: NUMBER
		exam: EXAM
		student: STUDENT
		answers: LIST OF ANSWER
		isTimeExpired: BOOLEAN

	CONSTRUCTOR(ex: EXAM, st: STUDENT)
		SET ex TO EXAM
		SET st TO STUDENT
        END CONSTRUCTOR

	METHODS
		GET_id()
			DISPLAY id
		END METHOD

		ADDANSWER(newAnswer: ANSWER)
			IF isTimeExpired EQUALS FALSE THEN
				ADD newAnswer TO answers
			ELSE
				DISPLAY "The time has been expreied"
			END IF
		END METHOD

		SET_EXPIRE()
			SET isTimeExpired TO TURE
		END METHOD

		CALCULATE_SCORE()
			SET SCORE TO 0
			REPEAT FROM I = 1 TO SIZE OF answers
				IF answers[I].selectedAnswer EQUALS answers[I].question.correctAnswer
					SET SCORE TO + answers[I].question.points
				END IF
			END REPEAT
		END METHOD
END CLASS

CLASS ANSWER
	ATTRIBUTES
		id: NUMBER
		question: QUESTION
		selectedAnswer: TEXT

	CONSTRUCTOR(q: QUESTION, selA: TEXT)
		SET q TO question
		SET selA TO selectedAnswer
        END CONSTRUCTOR

	METHODS
		GET_id()
			DISPLAY id
		END METHOD

		GET_selectedAnswer()
			DISPLAY selectedAnswer
		END METHOD
END CLASS

CLASS 
	ATTRIBUTES

		memberName: TEXT
		phone: NUMBER
		plan: MEMBERSHIPPLAN
		bookings: LIST OF BOOKING
	
		
	CONSTRUCTOR(mName: TEXT, p: NUMBER, planSh: MEMBERSHIPPLAN)
 		SET mName TO memberName
		SET p TO phone
		SET planSh TO plan
    	END CONSTRUCTOR


	METHODS
		GET_id()
			DISPLAY id
		END METHOD

		GET_memberName()
			DISPLAY memberName
		END MEHTHOD

		GET_phone()
			DISPLAY phone
		END METHOD

		ADDBOOKING(newBooking: BOOKING)
			IF GETREMAININGHOURS() GREATER THAN OR EQUAL TO newBooking.duration THEN
                		ADD newBooking TO bookings
            		ELSE
                		DISPLAY "Monthly quota rejected"
            		END IF
		END METHOD
		
		GETREMAININGHOURS()
            		SET usedHours TO 0
            		REPEAT FROM I = 1 TO SIZE OF bookings
               		SET usedHours TO usedHours + bookings[I].duration
            		END REPEAT
            		DISPLAY plan.monthlyQuota - usedHours
        	END METHOD
		
END CLASS

CLASS ROOM
	ATTRIBUTES
		id: NUMBER
		roomName: TEXT
		capacity: NUMBER
		bookings: LIST OF BOOKING

	CONSTRUCTOR(rn: TEXT, c: NUMBER)
 		SET rn TO roomName
		SET c TO capacity
    	END CONSTRUCTOR

	METHODS
		GET_id()
			DISPLAY id
		END METHOD

		GET_roomName()
			DISPLAY roomName
		END METHOD

		GET_capacity()
			DISPLAY capacity
		END METHOD

		ADDBOOKING(newBooking: BOOKING)
			SET overlap TO FALSE
			REPEAT FROM I = 1 TO SIZE OF bookings
				IF bookings[I].bookingDate EQUALS newBooking.bookingDate AND bookings[I].startTime EQUALS newBooking.startTime THEN
                    	SET overlap TO TRUE
                END IF
            END REPEAT
            	IF overlap EQUALS FALSE THEN
                	ADD newBooking TO bookings
            	ELSE
                	DISPLAY "Room already booked for this time"
            	END IF
END CLASS
