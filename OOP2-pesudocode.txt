Task 288: Bookshop

- A bookshop sells many books.
- A book belongs to one genre.
- A genre has many books.
- A customer has many sales.
- A sale belongs to one customer.
- A sale is for one book.
- A bookshop records many sales.


Blueprint:

CLASS BOOKAHOP
	ATTRIBUTES
        	id: NUMBER
        	shopName: TEXT
        	books: LIST OF BOOK
        	sales: LIST OF SALE
	

	CONSTRUCTOR(shopName: TEXT)
        	SET THIS.shopName TO shopName
    	END CONSTRUCTOR	


	METHODS
		GET_id()
            		DISPLAY id
        	END METHOD

		GET_shopName()
            		DISPLAY shopName
        	END METHOD

		ADDBOOK(newBook: BOOK)
            		ADD newBook TO books
        	END METHOD

		ADDSALE(newSale: SALE)
            		ADD newSale TO sales
        	END METHOD
END CLASS

CLASS CUSTOMER

    ATTRIBUTES
        id: NUMBER
        fullName: TEXT
        phone: TEXT
        purchaseHistory: LIST OF SALE

    CONSTRUCTOR(fullName: TEXT, phone: TEXT)
        SET THIS.fullName TO fullName
        SET THIS.phone TO phone
    END CONSTRUCTOR

    METHODS

	GET_id()
            DISPLAY id
        END METHOD

        GET_fullName()
            DISPLAY fullName
        END METHOD

        ADDSALE(sale: SALE)
            ADD sale TO purchaseHistory
        END METHOD

        GETPURCHASECOUNT()
            DISPLAY SIZE OF purchaseHistory
        END METHOD
END CLASS

CLASS SALE

    ATTRIBUTES
        id: NUMBER
        soldPrice: NUMBER
        saleDate: TEXT
        customer: CUSTOMER
        book: BOOK

    CONSTRUCTOR(soldPrice: NUMBER, saleDate: TEXT, customer: CUSTOMER, book: BOOK)
        IF soldPrice GREATER THAN 0 THEN
            SET THIS.soldPrice TO soldPrice
        ELSE
            DISPLAY "Invalid price"
        END IF
        SET THIS.saleDate TO saleDate
        SET THIS.customer TO customer
        SET THIS.book TO book
    END CONSTRUCTOR

    METHODS
        GET_id()
            DISPLAY id
        END METHOD

        GET_soldPrice()
            DISPLAY soldPrice
        END METHOD

        GET_saleDate()
            DISPLAY saleDate
        END METHOD

        PRINTRECEIPT()
            DISPLAY "Customer Full Name: " + customer.fullName
            DISPLAY "Book Title: " + book.title
            DISPLAY "Book Price: " + soldPrice
            DISPLAY "Sold Date: " + saleDate
        END METHOD

END CLASS
