Task 293: Shape Drawing App  


Blueprint:

CLASS SHAPE
	ATTRIBUTES
        	id: NUMBER
		color: TEXT
		position_x: NUMBER
		position_y: NUMBER
		
    	CONSTRUCTOR(color: TEXT, position_x: NUMBER, position_y: NUMBER)
		SET THIS.color TO color
		SET THIS.position_x TO NUMBER
		SET THIS.position_y TO NUMBER
    	END CONSTRUCTOR

    	METHODS
		GET_id()
			DISPLAY id
		END METHOD

		GET_color()
			DISPLAY color
		END METHOD

		GET_AREA()
			DISPLAY "Find the area: "
		END METHOD
			
		GET_PERIMETER()
			DISPLAY "Find the perimeter: "
		END METHOD
END CLASS 

CLASS RECTANGLE INHERITS SHAPE
	ATTRIBUTES
		width: NUMBER
		height: NUMBER
    	CONSTRUCTOR(color: TEXT, width: NUMBER, height: NUMBER)
		SUPER (color)
		SET THIS.width TO width
		SET THIS.height TO height
    	END CONSTRUCTOR

    	METHODS
		OVERRID GET_PERIMETER()
			SET PerimeterOfRectangle TO 2 * (width + height)
			DISPLAY PerimeterOfCircle
		END METHOD

		OVERRID GET_AREA()
			SET AreaOfRectangle TO width * height
			DISPLAY AreaOfRectangle
		END METHOD
END CLASS


CLASS CIRCLE INHERITS SHAPE
	ATTRIBUTES
 		radius: NUMBER

    	CONSTRUCTOR(color: TEXT, position_x: NUMBER, position_y: NUMBER, radius: NUMBER)
		SUPER(color, position_x, position_y)
		SET THIS.radius TO radius
    	END CONSTRUCTOR

    	METHODS
		OVERRID GET_PERIMETER()
			SET PerimeterOfCircle TO 2 * 3.14 * radius
			DISPLAY PerimeterOfCircle
		END METHOD

		OVERRID GET_AREA()
			SET AreaOfCircle TO 3.14 * radius * radius
			DISPLAY AreaOfCircle
		END METHOD
END CLASS

CLASS TRIANGLE INHERITS SHAPE
	ATTRIBUTES
		base: NUMBER
		height: NUMBER
		side1 : NUMBER
		side2: NUMBER
		side3 : NUMBER
 
    CONSTRUCTOR(color: TEXT, position_x: NUMBER, position_y: NUMBER, base: NUMBER, height: NUMBER, side1: NUMBER, side2: NUMBER, side3: NUMBER)
		SUPER (color, position_x, position_y)
		SET THIS.base TO base
		SET THIS.height TO height
		SET THIS.side1 TO side1
		SET THIS.side2 TO side2
		SET THIS.side3 TO side3
    END CONSTRUCTOR

    	METHODS
		OVERRID GET_PERIMETER()
			SET PerimeterOfTriangle TO side1 + side2 + side3
			DISPLAY PerimeterOfTriangle
		END METHOD

		OVERRID GET_AREA()
			SET AreaOfTriangle TO  (base * height) / 2
			DISPLAY AreaOfTriangle
		END METHOD

END CLASS

CLASS CANVAS
	ATTRIBUTES
		shapes: LIST OF SHAPE

	METHODS
		ADDSHAPE(newShape: SHAPE)
			ADD newShape TO shapes
		END METHOD

		REMOVESHAPE(removeShape: SHAPE)
			REMOVE removeShape FROM shapes
		END METHOD

		DIPLSAYSHAPES()
			DISPLAY shapes
		END METHOD
END CLASS


Pseudocode:
START 
	BEGIN
		DECLARE canvas AS CANVAS
		CREATE OBJECT shape1 OF RECTANGLE("Purple", 40, 10, 5, 8)
		CREATE OBJECT shape2 OF TRIANGLE("Blue", 15, 25, 4)
		CREATE OBJECT shape3 OF CIRCLE("Green", 5, 10, 6, 4, 5,5, 6)
		canvas.ADDSHAPE(shape1)
		canvas.ADDSHAPE(shape2)
		canvas.ADDSHAPE(shape3)
		shape1.GET_AREA()
		shape1.GET_PERIMETER()
		shape2.GET_AREA()
		shape2.GET_PERIMETER()
		shape3.GET_AREA()
		shape3.GET_PERIMETER()
		canvas.DIPLSAYSHAPES()
		
	STOP
END

Task 294: Employee Payroll  

CLASS EMPLOYEE
	ATTRIBUTES
		EmpId: NUMBER
		EmpName: TEXT
		Emp_HireDate: TEXT
		
	CONSTRUCTOR(EmpName: TEXT,Emp_HireDate: TEXT)
		SET THIS.EmpName TO EmpName
		SET THIS.Emp_HireDate TO Emp_HireDate
    	END CONSTRUCTOR

	METHODS	
		GET_id()
			DIPLSAY EmpId
		END METHOD

		GET_name()
			DISPLAY EmpName
		END METHOD
		calculatePay()
			DISPLAY "The payment of the employee is: "
		END METHOD
END CLASS

CLASS SalariedEmployee INHERITS EMPLOYEE

	ATTRIBUTES
		monthlyAmount:NUMBER

	CONSTRUCTOR(EmpName: TEXT,Emp_HireDate: TEXT,monthlyAmount:NUMBER)
		SUPER (EmpName,Emp_HireDate)
		SET THIS.monthlyAmount TO monthlyAmount
    	END CONSTRUCTOR

		
	METHODS	
		OVERRID calculatePay()
			DISPLAY "The payment of the employee is: " + monthlyAmount
		END METHOD
END CLASS

CLASS HourlyEmployee INHERITS EMPLOYEE
	ATTRIBUTES
		hourWorked : NUMBER
		hourlyRate: NUMBER
	CONSTRUCTOR(EmpName: TEXT,Emp_HireDate: TEXT, hourWorked : NUMBER, hourlyRate: NUMBER)
		SUPER (EmpName,Emp_HireDate)
		SET THIS.hourWorked TO hourWorked
		SET THIS.hourlyRate TO hourlyRate

    	END CONSTRUCTOR
		OVERRID calculatePay()
			SET monthlyAmount TO hourWorked * hourlyRate
			DISPLAY "The payment of the employee is: " + monthlyAmount
		END METHOD
		
	METHODS	
END CLASS

CLASS CommissionEmployee INHERITS EMPLOYEE
	ATTRIBUTES
		basePay: NUMBER
		percentageSales: NUMBER
		commissionRate: NUMBER

	CONSTRUCTOR(EmpName: TEXT,Emp_HireDate: TEXT, basePay: NUMBER, percentageSales: NUMBER, commissionRate:NUMBER)
		SUPER (EmpName,Emp_HireDate)
		SET THIS.basePay TO basePay
		SET THIS.percentageSales TO percentageSales
		SET THIS.commissionRate TO commissionRate
    	END CONSTRUCTOR

		
	METHODS	
		OVERRID calculatePay()
			SET monthlyAmount TO basePay + (percentageSales*commissionRate)
			DISPLAY "The payment of the employee is: " + monthlyAmount
		END METHOD
END CLASS

CLASS PAYROLL
	ATTRIBUTES
		employees : LIST OF EMPLOYEE

	CONSTRUCTOR()
        	SET THIS.employees TO EMPTY LIST
    	END CONSTRUCTOR

		
	METHODS	

		ADDEMPLOYEE(newEmp: EMPLOYEE)
			ADD newEmp TO employees
		END METHOD

		REMOVEEMPLOYEE(Emp : EMPLOYEE)
			REMOVE Emp FROM employees
		END METHOD

		DISPLAYEMPLOYEE()
			DISPLAY employees
		END METHOD
END CLASS


Pseudocode:
START
	BEGIN
		DECLARE Payrolls AS PAYROLL
		CREATE OBJECT salEmployee OF SalariedEmployee("Rashid Hamed", "01-Jul-2026", 1550)
		CREATE OBJECT HouEmployee OF HourlyEmployee("Khalfan Hamed", "01-Jan-2026", 12, 165)
		CREATE OBJECT comEmployee OF CommissionEmployee("Jokha Hamed", "28-Jun-2026", 1003, 0.10, 0.50)
		salEmployee.calculatePay()
		HouEmployee.calculatePay()
		comEmployee.calculatePay()
		Payrolls.ADDEMPLOYEE(HouEmployee)
		Payrolls.ADDEMPLOYEE(salEmployee)
		Payrolls.ADDEMPLOYEE(comEmployee)
		Payrolls.DISPLAYEMPLOYEE()	
	STOP
END
