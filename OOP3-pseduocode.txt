CLASS DEBITCARD INHERITS CARD
    ATTRIBUTES
       availableBalance: NUMBER

    CONSTRUCTOR(cardNumber: NUMBER, expiryDate: TEXT, status: TEXT, chip : CHIP, availableBalance: NUMBER)
        SUPER(cardNumber, expiryDate, status, chip)
        SET THIS.availableBalance TO availableBalance
    END CONSTRUCTOR

    METHODS
        OVERRIDE AUTHORIZEPURCHASE(newAmount: NUMBER)
		IF availableBalance GREATER THAN newAmount THEN
			DISPLAY "Purchase approved"
		ELSE
			DISPLAY "Not enough balance"
		END IF
	END METHOD
END CLASS

CLASS CREDITCARD INHERITS CARD
    ATTRIBUTES
       creditLimit: NUMBER
	outstandingBalance: NUMBER

    CONSTRUCTOR(cardNumber: NUMBER, expiryDate: TEXT, status: TEXT, chip : CHIP, outstandingBalance: NUMBER, creditLimit: NUMBER)
        SUPER(cardNumber, expiryDate, status, chip)
	SET THIS.outstandingBalance TO 0
        SET THIS.creditLimit TO creditLimit
    END CONSTRUCTOR

    METHODS
        OVERRIDE AUTHORIZEPURCHASE(newAmount: NUMBER)
		IF outstandingBalance + newAmount LESS THAN creditLimit THEN
			SET outstandingBalance TO outstandingBalance + newAmount
			DISPLAY "Purchase approved"
		ELSE
			DISPLAY "Credit has exceed the limit"
		END IF
	END METHOD

	GET_OUTSTANDINGBALANCE()
            DISPLAY outstandingBalance
        END METHOD

END CLASS
