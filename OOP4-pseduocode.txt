CLASS STORE
    ATTRIBUTES
	storeId: NUMBER
	storeName: TEXT
	instruments: LIST OF INSTURMENT

    CONSTRUCTOR(storeName: TEXT)
 	SET THIS.storeName TO storeName
	SET THIS.instruments TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
 	GET_storeId()
		DISPLAY storeId
	END METHOD

	GET_storeName()
		DISPLAY storeName
	END METHOD

	ADDINSTURMENT(newInstrument : INSTURMENT)
		ADD newInstrument TO instruments
	END METHOD

	REMOVEINSTURMENT(instrument : INSTURMENT)
		REMOVE instrument FROM instruments
	END METHOD

	DISPLAYINSTURMENT()
    	FOR EACH instrument IN instruments
	        instrument.GET_brandInstrument()
        	instrument.GET_priceInstrument()
        	instrument.describePlayStyle()
    	END FOR
	END METHOD
END CLASS

CLASS STANDARDRIDE INHERITS RIDE
    ATTRIBUTES
	distance: NUMBER
	baseKilometer: NUMBER

    CONSTRUCTOR(pickupLocation: TEXT, dropOffLocation: TEXT, route: ROUTE, distance: NUMBER, baseKilometer: NUMBER)
	SUPER(pickupLocation, dropOffLocation, route)
	SET THIS.distance TO distance
	SET THIS.baseKilometer TO baseKilometer
    END CONSTRUCTOR

    METHODS
	OVERRIDE calculateFare()
		SET fare TO distance * baseKilometer
		DISPLAY "The standard ride fare is:" + fare
	END METHOD
END CLASS

CLASS SHAREDRIDE INHERITS STANDARDRIDE
    ATTRIBUTES
	passengers: NUMBER

    CONSTRUCTOR(pickupLocation: TEXT, dropOffLocation: TEXT, route: ROUTE, distance: NUMBER, baseKilometer: NUMBER, passengers: NUMBER)
	SUPER(pickupLocation, dropOffLocation, route, distance, baseKilometer)
	SET THIS.passengers TO passengers
    END CONSTRUCTOR

    METHODS
	 OVERRIDE calculateFare()
