CLASS VECHICLECLAIM INHERITS CLAIM
	ATTRIBUTES
		repairEstimates: NUMBER
		fixedExcess : NUMBER
		
    	CONSTRUCTOR(claimNo: NMUMBER, dateFiled: TEXT, claimStatus: TEXT, documentationFolder: DOCUMENTATIONFOLDER, repairEstimates: NUMBER, fixedExcess : NUMBER)
		SUPER(claimNo, dateFiled, claimStatus, documentationFolder)
		SET THIS.repairEstimates TO repairEstimates
		SET THIS.fixedExcess TO fixedExcess
    	END CONSTRUCTOR

    	METHODS
		OVERRIDE calculatePayout()
			SET PAYOUT TO repairEstimates - fixedExcess
			DISPLAY "The claim payout is: " + PAYOUT
		END METHOD
END CLASS

CLASS MEDICALCLAIM INHERITS CLAIM
	ATTRIBUTES
		treatmentCosts: NUMBER
		coverageLimit : NUMBER
		
    	CONSTRUCTOR(claimNo: NMUMBER, dateFiled: TEXT, claimStatus: TEXT, documentationFolder: DOCUMENTATIONFOLDER, treatmentCosts: NUMBER, coverageLimit : NUMBER)
		SUPER(claimNo, dateFiled, claimStatus, documentationFolder)
		SET THIS.treatmentCosts TO treatmentCosts
		SET THIS.coverageLimit TO coverageLimit
    	END CONSTRUCTOR

    	METHODS
		OVERRIDE calculatePayout()
			IF treatmentCosts GREATER THAN coverageLimit THEN
				SET PAYOUT TO coverageLimit
			ELSE 
				SET PAYOUT TO treatmentCosts
			END IF 
			DISPLAY "The medical claim payout is: " + PAYOUT
		END METHOD
END CLASS

CLASS PROPERTYCLAIM INHERITS CLAIM
	ATTRIBUTES
		estimatedDamage: NUMBER

    	CONSTRUCTOR(claimNo: NMUMBER, dateFiled: TEXT, claimStatus: TEXT, documentationFolder: DOCUMENTATIONFOLDER, estimatedDamage: NUMBER, percentage : NUMBER)
		SUPER(claimNo, dateFiled, claimStatus, documentationFolder)
		SET THIS.percentage TO percentage 
		SET THIS.estimatedDamage TO estimatedDamage
    	END CONSTRUCTOR

    	METHODS
		OVERRIDE calculatePayout()
            		SET PAYOUT TO estimatedDamage - (estimatedDamage * percentage / 100)
            		DISPLAY "The property claim payout is: " + PAYOUT
       		END METHOD
END CLASS

CLASS HOURLYINVOICE INHERITS INVOICE
	ATTRIBUTES
		loggedHours: NUMBER
		hourlyRate: NUMBER
		approved : BOOLEAN
		
	CONSTRUCTOR(inoviceNo: NUMBER, issueDate: TEXT, clientName: TEXT, paymentTerms: PAYMENTTERMS, loggedHours: NUMBER, hourlyRate: NUMBER, approved : BOOLEAN)
		SUPER(inoviceNo, issueDate, clientName, paymentTerms)
		SET THIS.loggedHours TO loggedHours
		SET THIS.hourlyRate TO hourlyRate
		SET THIS.approved TO approved
    	END CONSTRUCTOR
		
	METHODS
 		OVERRIDE calculateTotalAmount()
			IF approved EQUALS TO TRUE THEN
				SET TOTAL TO loggedHours * hourlyRate
				DISPLAY "The total amount due is: " + TOTAL
			ELSE 
				SET TOTAL TO 0
				DISPLAY "The hour entry that wasn't first marked as approved by the client"
			END IF  
		END METHOD

END CLASS


CLASS RETAINERINVOICE INHERITS FIXEDPRICEINVOICE
	ATTRIBUTES
		charges: NUMBER
		hourAllowance: NUMBER
		hourStyle: NUMBER
		
	CONSTRUCTOR(inoviceNo: NUMBER, issueDate: TEXT, clientName: TEXT, paymentTerms: PAYMENTTERMS, singleProject: NUMBER, charges: NUMBER, hourAllowance: NUMBER, hourStyle: NUMBER)
		SUPER(inoviceNo, issueDate, clientName, paymentTerms, singleProject)
		SET THIS.charges TO charges
		SET THIS.hourAllowance TO hourAllowance
		SET THIS.hourStyle TO hourStyle
    	END CONSTRUCTOR
		
	METHODS
 		OVERRIDE calculateTotalAmount()
			SET extraHours TO charges - hourAllowance
			IF extraHours LESS THAN 0 THEN
    			SET extraHours TO 0
			END IF
			SET TOTAL TO singleProject + (extraHours * hourStyle)
			DISPLAY "The total amount due is: " + TOTAL
		END METHOD
END CLASS

CLASS WEBSERVER INHERITS SERVER
    ATTRIBUTES
        activeConnections: NUMBER
        maxConnections: NUMBER

    CONSTRUCTOR(serialNumber: TEXT, uptime: NUMBER, hardwareSpecification: HARDWARESPECIFICATION, activeConnections: NUMBER, maxConnections: NUMBER)
        SUPER(serialNumber, uptime, hardwareSpecification)
        SET THIS.activeConnections TO activeConnections
        SET THIS.maxConnections TO maxConnections
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateLoad()
            SET load TO (activeConnections / maxConnections) * 100
            DISPLAY "The web server load is: " + load
        END METHOD

CLASS DATABASESERVER INHERITS SERVER
    ATTRIBUTES
        queryThroughput: NUMBER
        maxQueries: NUMBER

    CONSTRUCTOR(serialNumber: TEXT, uptime: NUMBER, hardwareSpecification: HARDWARESPECIFICATION, queryThroughput: NUMBER, maxQueries: NUMBER)
        SUPER(serialNumber, uptime, hardwareSpecification)
        SET THIS.queryThroughput TO queryThroughput
        SET THIS.maxQueries TO maxQueries
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateLoad()
            SET load TO (queryThroughput / maxQueries) * 100

            IF queryThroughput GREATER THAN OR EQUAL TO maxQueries THEN
                DISPLAY "Server is at full capacity. No new connections accepted."
            END IF

            DISPLAY "The database server load is: " + load
        END METHOD
END CLASS


CLASS CACHESERVER INHERITS DATABASESERVER
    ATTRIBUTES
        cacheHitRatio: NUMBER

    CONSTRUCTOR(serialNumber: TEXT, uptime: NUMBER, hardwareSpecification: HARDWARESPECIFICATION, queryThroughput: NUMBER, maxQueries: NUMBER, cacheHitRatio: NUMBER)
        SUPER(serialNumber, uptime, hardwareSpecification, queryThroughput, maxQueries)
        SET THIS.cacheHitRatio TO cacheHitRatio
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateLoad()
            SET load TO (queryThroughput / maxQueries) * 100
            SET effectiveLoad TO load - (load * cacheHitRatio / 100)
            DISPLAY "The cache server effective load is: " + effectiveLoad
        END METHOD
END CLASS
