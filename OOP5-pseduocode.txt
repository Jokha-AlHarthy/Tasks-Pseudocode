Task 308: Insurance Claims Processing 

Blueprint:

CLASS CLAIM
    ATTRIBUTES
	claimId: NUMBER
	claimNo: NMUMBER
	dateFiled: TEXT
	claimStatus: TEXT
	documentationFolder: DOCUMENTATIONFOLDER

    CONSTRUCTOR(claimNo: NMUMBER, dateFiled: TEXT, claimStatus: TEXT, documentationFolder: DOCUMENTATIONFOLDER)
	SET THIS.claimNo TO claimNo
	SET THIS.dateFiled TO dateFiled
	SET THIS.claimStatus TO claimStatus
	SET THIS.documentationFolder TO documentationFolder
    END CONSTRUCTOR

    METHODS
	GET_id()
		DISPLAY claimId
	END METHOD

	GET_claimNo()
		DISPLAY claimNo
	END METHOD

	GET_dateFiled()
		DIPLSAY dateFiled
	END METHOD

	GET_claimStatus()
		DISPLAY claimStatus
	END METHOD
	
	calculatePayout()
		DISPLAY "The claim payout is: "
	END METHOD
	
END CLASS

CLASS VECHICLECLAIM INHERITS CLAIM
	ATTRIBUTES
		repairEstimates: NUMBER
		fixedExcess : NUMBER
		
    	CONSTRUCTOR(claimNo: NMUMBER, dateFiled: TEXT, claimStatus: TEXT, documentationFolder: DOCUMENTATIONFOLDER, repairEstimates: NUMBER, fixedExcess : NUMBER)
		SUPER(claimNo, dateFiled, claimStatus, documentationFolder)
		SET THIS.repairEstimates TO repairEstimates
		SET THIS.fixedExcess TO fixedExcess
    	END CONSTRUCTOR

    	METHODS
		OVERRIDE calculatePayout()
			SET PAYOUT TO repairEstimates - fixedExcess
			DISPLAY "The claim payout is: " + PAYOUT
		END METHOD
END CLASS

CLASS MEDICALCLAIM INHERITS CLAIM
	ATTRIBUTES
		treatmentCosts: NUMBER
		coverageLimit : NUMBER
		
    	CONSTRUCTOR(claimNo: NMUMBER, dateFiled: TEXT, claimStatus: TEXT, documentationFolder: DOCUMENTATIONFOLDER, treatmentCosts: NUMBER, coverageLimit : NUMBER)
		SUPER(claimNo, dateFiled, claimStatus, documentationFolder)
		SET THIS.treatmentCosts TO treatmentCosts
		SET THIS.coverageLimit TO coverageLimit
    	END CONSTRUCTOR

    	METHODS
		OVERRIDE calculatePayout()
			IF treatmentCosts GREATER THAN coverageLimit THEN
				SET PAYOUT TO coverageLimit
			ELSE 
				SET PAYOUT TO treatmentCosts
			END IF 
			DISPLAY "The medical claim payout is: " + PAYOUT
		END METHOD
END CLASS

CLASS PROPERTYCLAIM INHERITS CLAIM
	ATTRIBUTES
		eastimatedDamage: NUMBER
		percentage : NUMBER
		
    	CONSTRUCTOR(claimNo: NMUMBER, dateFiled: TEXT, claimStatus: TEXT, documentationFolder: DOCUMENTATIONFOLDER, eastimatedDamage: NUMBER, percentage : NUMBER)
		SUPER(claimNo, dateFiled, claimStatus, documentationFolder)
		SET THIS.percentage TO percentage 
		SET THIS.eastimatedDamage TO eastimatedDamage
    	END CONSTRUCTOR

    	METHODS
		OVERRIDE calculatePayout()
            		SET PAYOUT TO eastimatedDamage - (eastimatedDamage * percentage / 100)
            		DISPLAY "The property claim payout is: " + PAYOUT
       		END METHOD
END CLASS

CLASS POLICY
    ATTRIBUTES
        policyId: NUMBER
        policyName: TEXT
        claims: LIST OF CLAIM

    CONSTRUCTOR(policyName: TEXT)
        SET THIS.policyName TO policyName
    END CONSTRUCTOR

    METHODS
        GET_id()
            DISPLAY policyId
        END METHOD

        GET_policyName()
            DISPLAY policyName
        END METHOD

        ADDCLAIM(newClaim: CLAIM)
            ADD newClaim TO claims
        END METHOD

        REMOVECLAIM(claim: CLAIM)
            REMOVE claim FROM claims
        END METHOD

        DISPLAYCLAIMS()
            DISPLAY claims
        END METHOD
END CLASS


pseudocode:
START
    BEGIN

        CREATE OBJECT documentationFolder OF DOCUMENTATIONFOLDER("Claim Documents")
        CREATE OBJECT document1 OF DOCUMENT("Repair Invoice", "PDF")
        CREATE OBJECT document2 OF DOCUMENT("Insurance Form", "Form")
        documentationFolder.ADDDOCUMENT(document1)
        documentationFolder.ADDDOCUMENT(document2)
        CREATE OBJECT policy OF POLICY("Vehicle Policy")
        CREATE OBJECT vehicleClaim OF VEHICLECLAIM(1001, "01-Aug-2026", "Pending", documentationFolder, 5000, 500)
        CREATE OBJECT medicalClaim OF MEDICALCLAIM(1002,"02-Aug-2026", "Approved", documentationFolder, 2500, 2000)
        CREATE OBJECT propertyClaim OF PROPERTYCLAIM(1003,"03-Aug-2026", "Pending", documentationFolder, 10000, 20)
        policy.ADDCLAIM(vehicleClaim)
        policy.ADDCLAIM(medicalClaim)
        policy.ADDCLAIM(propertyClaim)
        vehicleClaim.calculatePayout()
        medicalClaim.calculatePayout()
        propertyClaim.calculatePayout()
        policy.DISPLAYCLAIMS()
    END
STOP

Task 309: University Library Loans (Medium)

Blueprint:

CLASS LOANITEM
	ATTRIBUTES
		loanId: NUMBER
		loanTitle: TEXT
		barcode: TEXT
		loanDuration: NUMBER
		conditionReport: CONDITIONREPORT

	CONSTRUCTOR(loanTitle: TEXT, barcode: TEXT, loanDuration: NUMBER, conditionReport: CONDITIONREPORT)
		SET THIS.loanTitle TO loanTitle
		SET THIS.barcode TO barcode
		SET THIS.loanDuration TO loanDuration
		SET THIS.conditionReport TO conditionReport
    	END CONSTRUCTOR

    	METHODS
		GET_id()
			DISPLAY loanId
		END METHOD

		GET_loanTitle()
			DISPLAY loanTitle
		END METHOD

		GET_barcode()
			DISPLAY barcode
		END METHOD
	
		GET_loanDuration()
			DISPLAY loanDuration
		END METHOD

		calculateLateFee()
			DISPLAY "The late fee is: "
		END METHOD
END CLASS


CLASS REFERNCETEXTBOOK INHERITS LOANITEM
	ATTRIBUTES
		higherLateFee: NUMBER
		daysOverDue: NUMBER

	CONSTRUCTOR(loanTitle: TEXT, barcode: TEXT, loanDuration: NUMBER, conditionReport: CONDITIONREPORT, higherLateFee: NUMBER, daysOverDue: NUMBER)
		SUPER(loanTitle, barcode, loanDuration,conditionReport)
		SET THIS.higherLateFee TO higherLateFee
		SET THIS.daysOverDue TO daysOverDue
    	END CONSTRUCTOR

	METHODS
		OVERRIDE calculateLateFee()
			SET lateFee TO daysOverDue * higherLateFee
			DISPLAY "The late fee is: " + lateFee
		END METHOD
    	
END CLASS

CLASS JOURNALISSUE INHERITS REFERNCETEXTBOOK
	ATTRIBUTES
		renewable: BOOLEAN

	CONSTRUCTOR(loanTitle: TEXT, barcode: TEXT, loanDuration: NUMBER,conditionReport: CONDITIONREPORT, higherLateFee: NUMBER, daysOverDue: NUMBER, renewable: BOOLEAN)
    		SUPER(loanTitle, barcode, loanDuration, conditionReport, higherLateFee, daysOverDue)
    		SET THIS.renewable TO renewable
	END CONSTRUCTOR
	
	METHODS
		OVERRIDE calculateLateFee()
			SET lateFee TO daysOverDue * higherLateFee
			SET renewable TO FALSE
			DISPLAY "The late fee is: " + lateFee
			DISPLAU "The renewable status is: " + renewable
		END METHOD 
    	
END CLASS

CLASS STUDENTACCOUNT
    ATTRIBUTES
        accountId: NUMBER
        studentName: TEXT
        loanItems: LIST OF LOANITEM

    CONSTRUCTOR(studentName: TEXT)
        SET THIS.studentName TO studentName
    END CONSTRUCTOR

    METHODS
        GET_id()
            DISPLAY accountId
        END METHOD

        GET_studentName()
            DISPLAY studentName
        END METHOD

        ADDLOANITEM(newLoanItem: LOANITEM)
            ADD newLoanItem TO loanItems
        END METHOD

        REMOVELOANITEM(loanItem: LOANITEM)
            REMOVE loanItem FROM loanItems
        END METHOD

        DISPLAYLOANITEMS()
            DISPLAY loanItems
        END METHOD
END CLASS


pseudocode:
START
    BEGIN

        CREATE OBJECT conditionReport OF CONDITIONREPORT("Good", "No Damage")
        CREATE OBJECT studentAccount OF STUDENTACCOUNT("Ahmed")
        CREATE OBJECT standardBook OF STANDARDBOOK("Java Programming", "BK1001", 14, conditionReport, 0.50, 3)
        CREATE OBJECT referenceTextbook OF REFERENCETEXTBOOK("Operating Systems", "RF2001", 1, conditionReport, 3.00, 2)
        CREATE OBJECT journalIssue OF JOURNALISSUE("IEEE Journal", "JR3001", 1, conditionReport, 3.00, 4, TRUE)
        studentAccount.ADDLOANITEM(standardBook)
        studentAccount.ADDLOANITEM(referenceTextbook)
        studentAccount.ADDLOANITEM(journalIssue)
        standardBook.calculateLateFee()
        referenceTextbook.calculateLateFee()
        journalIssue.calculateLateFee()
        studentAccount.DISPLAYLOANITEMS()
    END
STOP
