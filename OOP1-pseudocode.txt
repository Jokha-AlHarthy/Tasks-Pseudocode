Task 279: Movie Theater

- A film has many showtimes.
- A showtime belongs to one film.
- A showtime is assigned to one screen.
- A screen can have many showtimes.
- A showtime has many tickets.
- A ticket belongs to one showtime.

	END CONSTRCTOR

	METHODS
		ADDSHOWRIME(show: SHOWTIMES)
			ADD show TO showtimes

		GET_id()

		END METHOD

		GET_title()
			DISPLAY title
		END METHOD

		GET_duration()
			DISPLAY duration
		END METHOD

		SET_duration(newDuration: TEXT)
			IF LENGTH OF newDuration GREATHER THAN 0 THEN
				SET duration TO newDuration
			ELSE
				DISPLAY "Error assigning time duration"
			END IF
		END METHOD

		GET_caption()
			DISPLAY caption
		END METHOD

		GET_filmType()
			DISPLAY filmtype
		END METHOD

END CLASS

Task 280: Warehouse Inventory

- A product belongs to one vendor.
- A vendor supplies many products.
- A product has many storage bins.
- A product has many stock movements.
- A stock movement belongs to one product.

Blueprint:

CLASS PRODUCT
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		category: TEXT
		vendor : VENDOR
		bins: LIST OF STORAGEBIN
		movements: LIST OF STOCKMOVEMENT

	CONSTRCUTOR(n: TEXT, c: TEXT, v: VENDOR)
		SET name TO n
		SET category TO c
		SET vendor TO v
	END CONSTRCTOR
	
	METHODS
		GET_id()
			DISPLAY id
		END METHOD

		GET_name()
			DISPLAY name
		END METHOD

		SET_name(newName : TEXT)
			IF LENGTH OF newname GREATE THAN 0 THEN
				DISPLAY "Enter product new name: "
				SET name TO newname
			ELSE
				DISPLAY "Invalid name, please enter a name"
			END IF
		END METHOD

		AddBin(newBins : TEXT)
			ADD bin TO STORAGEBIN
		END METHOD

		ADDMOVEMENT(move : STOCKMOVEMENT)
			IF move.movementType EQUALS "OUT" THEN
				IF GET_totalstock() GREATER THEN move.qauntity THEN
					ADD move TO movements
				ELSE
					DISPLAY "Not enough stock"
				END IF
			ELSE
				ADD move TO movements
			END IF
		END METHOD
		
		GET_totalStock()
			SET total TO 0
			REAPET FROM I = 1 TO SIZE OF bins
				SET total TO total + bins[I].qaunity
			END REAPEAT
			DISPLAY total
		END METHOD		
END CLASS

CLASS STORAGEBIN
	ATTRIBUTES
		id: NUMBER
		location: TEXT
		quantity: NUMBER

	CONSTRCTOR(l: TEXT, q: NUMBER)
		SET l TO location
		SET q TO quantity
	END CONSTRUCTOR


	METHODS
		GET_location()
			DISPLAY location
		END METHOD
		
		GET_quantity
			DISPLAY quantity
		END METHOD

		SET_quantity(newQuantity : NUMBER)
			IF newQuantity GREATER THAN OR EQUALS TO 0 THEN
				SET quantity TO newQuantity
			ELSE
				DISPLAY "Invaild quantity"
			END IF
		END METHOD
END CLASS

CLASS VENDOR
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		contact: TEXT
		rating: NUMBER
		products: LIST OF PRODUCT

	CONSTRUCTOR(nv: TEXT, cv: TEXT)
		SET nv TO name
		SET cv TO contact
		SET rating TO 5
	END CONSTRCTOR

	
	METHODS
		ADDPRODUCT (product: PRODUCT)
			ADD product TO products
		END METHOD

		UPDATERATE(lateDilvery:BOOLEAN)
			IF lateDilvery EQUALS TRUE THEN
				SET rating TO rating -1
			END IF
		END METHOD

		GET_rating()
			DISPLAY rating
		END METHOD

END CLASS

CLASS STOCKMOVMENT
	ATTRIBUTES
		id: NUMBER
		movementType: TEXT
		quantity: NUMBER
		date: TEXT

	CONSTRUCTOR(ts : TEXT, qs: NUMBER, ds: TEXT)
		SET ts TO movementType
		SET qs TO quantity

	END CONTRUCTOR

	METHODS
		GET_quantityStock()
			DISPLAY quantity
		END METHOD

		END METHOD

END CLASS

Task 281: School Classroom

- A classroom has one teacher.
- A classroom has many students.
- A student can't be enrolled in two classes at the same scheduled time.
- A student has many grades.

Blueprint:

CLASS CLASSROOM
	ATTRIBUTES
		id: NUMBER
		className : TEXT
		subject: TEXT
		schedule : TEXT
		maxSize: NUMBER
		teacher: TEACHER
		students: LIST OF STUDENT
	
	CONTRACTOR(className: TEXT, subject: TEXT, schedule : TEXT, maxSize: NUMBER, teacher: TEACHER)
		SET THIS.className TO className
		SET THIS.subject TO subject
		SET THIS.schedule TO schedule
		SET THIS.maxSize TO maxSize
		SET THIS.teacher TO teacher
	END CONSTRCTOR
	
	METHODS
		ADDSTUDENT(newStudent: STUDENT)
			IF SIZE OF students LESS THAN maxSize THEN
				IF student.schedule NOT EQUAL TO schedule THEN
					ADD student TO students
				ELSE 
					DISPLAY "Student already has class at same time"
				END IF
			ELSE
				DISPLAY "Classroos is full"
			END IF
		END METHOD
		
		GET_id()
            		DISPLAY id
        	END METHOD

		GET_className()
            		DISPLAY className
        	END METHOD

		GET_subject()
            		DISPLAY subject
       		END METHOD
		
	
END CLASS

CLASS STUDENT
	ATTRIBUTES
		id: NUMBER
		fullName: TEXT
		schedule : TEXT
		grade: LIST OF GRADE

	CONSTRUCTOR(fullName : TEXT, schedule : TEXT)
		SET THIS.fullName TO fullName
        	SET THIS.schedule TO schedule
    	END CONSTRUCTOR

	METHODS
		ADDGRADE(grade : GRADE)
            		ADD grade TO grades
        	END METHOD

		GETAVERAGE()
			SET total TO 0
            		SET totalWeight TO 0
			REPEAT FROM I EQUALS 1 TO SIZE OF grades
                		SET total TO total + (grades[I].mark * grades[i].weight)
                		SET totalWeight TO totalWeight + grades[I].weight

            		END REPEAT
			DISPLAY total / totalWeight

        	END METHOD
		
		GET_fullName()
            		DISPLAY fullName
        	END METHOD
	
END CLASS
