		END METHOD

		GET_duration()

CLASS SHOWTIME
	ATTRIBUTES
		id: NUMBER
        	date: TEXT
        	time: TEXT
        	screen: SCREEN
        	tickets: LIST OF TICKET
	CONSTRUCTOR(date: TEXT, time: TEXT, screen: SCREEN)
        	SET THIS.date TO date
        	SET THIS.time TO time
        	SET THIS.screen TO screen
    	END CONSTRUCTOR

	METHODS
	
		ADDTICKET(ticket: TICKET)
    	SET booked TO FALSE
    	REPEAT FROM I = 1 TO SIZE OF tickets
        	IF tickets[I].seatNumber EQUALS ticket.seatNumber THEN
            	SET booked TO TRUE
        	END IF
    	END REPEAT
    	IF booked EQUALS FALSE THEN
        	ADD ticket TO tickets
    	ELSE
        	DISPLAY "Seat already booked"
    	END IF
		END METHOD
 
		GET_id()
			DISPLAY id
		END METHOD
	
		GET_date()
			DISPLAY date
		END METHOD

		SET_date(newDate: TEXT)
			DISPLAY "Enter date you want for the show"
			SET date TO newDate
		END METHOD()

		GET_time()
			DISPLAY time
		END METHOD

		SET_time(newTime: TEXT)
			DISPLAY "Enter time you want for the show"
			SET time TO newTime
		END METHOD()

		GET_screen()
			DISPLAY screen
		END METHOD

		DISPLAYSHOWTIME()
            DISPLAY id
            DISPLAY date
            DISPLAY time
        END METHOD

		GETREMAININGSEATS()
    		SET remaining TO screen.capacity - SIZE OF tickets
    		DISPLAY remaining
		END METHOD
		
END CLASS

CLASS PRODUCT
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		category: TEXT
		vendor : VENDOR
		bins: LIST OF STORAGEBIN
		movements: LIST OF STOCKMOVEMENT

	CONSTRUCTOR(n: TEXT, c: TEXT, v: VENDOR)
		SET name TO n
		SET category TO c
		SET vendor TO v
	END CONSTRUCTOR
	
	METHODS
		GET_id()
			DISPLAY id
		END METHOD

		GET_name()
			DISPLAY name
		END METHOD

		SET_name(newName : TEXT)
			IF LENGTH OF newname GREATE THAN 0 THEN
				DISPLAY "Enter product new name: "
				SET name TO newname
			ELSE
				DISPLAY "Invalid name, please enter a name"
			END IF
		END METHOD

		ADDBIN(bin : STORAGEBIN)
    		ADD bin TO bins
		END METHOD

		ADDMOVEMENT(move : STOCKMOVEMENT)
    		IF move.movementType EQUALS "OUT" THEN
        		IF GET_totalStock() GREATER THAN OR EQUAL TO move.quantity THEN
            		ADD move TO movements
        		ELSE
            		DISPLAY "Not enough stock"
        		END IF
    		ELSE
        		ADD move TO movements
    		END IF
		END METHOD
		
		GET_totalStock()
			SET total TO 0
			REPEAT FROM I = 1 TO SIZE OF bins
				SET total TO total + bins[I].quantity

		END METHOD

		id: NUMBER

		subject: TEXT
		schedule : TEXT
		maxSize: NUMBER
		teacher: TEACHER

		SET THIS.subject TO subject
		SET THIS.schedule TO schedule
		SET THIS.maxSize TO maxSize
		SET THIS.teacher TO teacher
	END CONSTRUCTOR
	
	METHODS
		ADDSTUDENT(newStudent: STUDENT)
			IF SIZE OF students LESS THAN maxSize THEN
				IF newStudent.schedule NOT EQUAL TO schedule THEN
    				ADD newStudent TO students
				ELSE 
					DISPLAY "Student already has class at same time"
				END IF
			ELSE
				DISPLAY "Classroom is full"
			END IF
		END METHOD
		
		GET_id()

       	END METHOD

        	SET THIS.schedule TO schedule
    	END CONSTRUCTOR

	METHODS
		ADDGRADE(grade : GRADE)
            		ADD grade TO grades
        END METHOD

		GETAVERAGE()
    		SET total TO 0

		END METHOD

		END METHOD

CLASS POLICY
	ATTRIBUTES
		id: NUMBER
		policyNumber: TEXT
		holder: POLICYHOLDER
		coverages: LIST OF COVERAGE
		claims: LIST OF CLAIM
		status: TEXT

	CONSTRUCTOR(policyNumber: TEXT, holder: POLICYHOLDER)
		SET THIS.policyNumber TO policyNumber
		SET THIS.holder TO holder
		SET THIS.status TO "Active"
	END CONSTRUCTOR

	METHODS
		ADDCOVERAGE(coverage: COVERAGE)
            		ADD coverage TO coverages
        	END METHOD

        	ADDCLAIM(claim: CLAIM)
            		ADD claim TO claims
        	END METHOD
		
		GETREMAININGCOVERAGE()
			SET totalCoverage TO 0
            		SET totalClaims TO 0

            		REPEAT FROM I = 1 TO SIZE OF coverages
               			SET totalCoverage TO totalCoverage + coverages[I].limit
            		END REPEAT

            		REPEAT FROM I = 1 TO SIZE OF claims
                		IF claims[I].status EQUALS "Approved" THEN
                    			SET totalClaims TO totalClaims + claims[I].amount
                		END IF
           		END REPEAT

          	  	DISPLAY totalCoverage - totalClaims
        	END METHOD

END CLASS

        	END METHOD

		id: NUMBER
		amount: NUMBER
		status: TEXT
		policy: POLICY

	CONSTRUCTOR(amount: NUMBER, status: TEXT, policy: POLICY)
		SET THIS.amount TO amount
    	SET THIS.status TO status
    	SET THIS.policy TO policy
    END CONSTRUCTOR

	METHODS
		SET_status(newStatus: TEXT)
    		IF status EQUALS "Pending" AND newStatus EQUALS "Under Review" THEN
        		SET status TO newStatus
    		ELSE IF status EQUALS "Under Review" THEN
        		SET status TO newStatus
    		ELSE
        		DISPLAY "Invalid status change"
    		END IF
		END METHOD

		GET_amount()
            	DISPLAY amount
        END METHOD

        GET_status()
    		DISPLAY status
		END METHOD
END CLASS

Task 286: Software Sprint

- A sprint has many tasks.
- A developer has one or many tasks.
- A team has many sprints.


Blueprint:


CLASS SPRINT
	ATTRIBUTES
		id: NUMBER
		sprintName: TEXT
		taskList: LIST OF TASKS

	CONSTRCTOR(sprintName: TEXT)
		SET THIS.sprintName TO sprintName
	END CONSTRCUTOR

	METHODS
		GET_id()
			DISPALY id
		END METHOD

		GET_Progress()
            		SET completeCount TO 0
            		SET totalTasks TO LENGTH(THIS.taskList)
            
            		IF totalTasks EQUALS 0 THEN
                		DISPLAY "Sprint Progress: 0%"
            		END IF

            		FOREACH t IN THIS.taskList
                		IF t.GET_Status() EQUALS "Done" THEN
                    			SET completeCount = completeCount + 1
                		END IF
            		END FOREACH

            		SET percentage TO (completeCount / totalTasks) * 100
            		DISPLAY "Sprint Progress: " + percentage + "%"
        	END METHOD


		ADDTASK(newTask: TEXT, dev: TEXT)
			ADD newTask TO taskList
		END METHOD	
END CLASS

CLASS TASK
    ATTRIBUTES
        id: NUMBER
        title: TEXT
        points: NUMBER
        status: TEXT         
        assignedDev: LIST OF DEVELOPER

    CONSTRUCTOR(title: TEXT, points: NUMBER)
        SET THIS.id TO id
        SET THIS.title TO title
        SET THIS.points TO points
        SET THIS.status TO "Todo"
        SET THIS.assignedDev TO NULL
    END CONSTRUCTOR

    METHODS
        GET_id()
            DISPLAY THIS.id
        END METHOD

        GET_Points()
            DISPLAY THIS.points
        END METHOD

        GET_Status()
            DISPLAY THIS.status
        END METHOD

        ASSIGN_Developer(dev: TEXT)
            SET assignedDev TO dev
        END METHOD

        GET_Developer()
            DISPLAY assignedDev
        END METHOD

        MARKSTATUS()
            IF status EQUALS "Todo" THEN
                SET status TO "In Progress"
            ELSEIF status EQULAS "In Progress" THEN
                SET status TO "Review"
            ELSEIF status EQULAS "Review" THEN
                SET status TO "Done"
            ELSE
                DISPLAY "Task is already completed"
            END IF
        END METHOD

        MARKCOMPLETE()
            IF status EQUALS "Review" THEN
                SET status TO "Done"
            ELSEIF EQUALS "Done" THEN
                DISPLAY "Task already done"
            ELSE
                DISPLAY "Error: Cannot complete task directly from: " + status
            END IF
        END METHOD
END CLASS

	ATTRIBUTES
        	id: NUMBER
        	devName: TEXT
        	maxCapacity: NUMBER
        

    CONSTRUCTOR(devName: TEXT, maxCapacity: NUMBER)
        SET THIS.id TO id
        SET THIS.devName TO devName

    END CONSTRUCTOR

	 METHODS

            		DISPLAY id
        	END METHOD

        	END METHOD
END CLASS

CLASS BACKLOG
    ATTRIBUTES
        id: NUMBER
        unassignedTasks: LIST OF TASK

    CONSTRUCTOR(id: NUMBER)
        SET THIS.id TO id
    END CONSTRUCTOR

    METHODS
        GET_id()
            DISPLAY id
        END METHOD

        ADDTOBACKLOG(newTask: TEXT)
            ADD newTask TO unassignedTasks
