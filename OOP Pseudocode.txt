	END CONTUCTOR

		END METHOD

Scenario 3: Restaurant Order

Blueprint:

CLASS ORDER
    ATTRIBUTES
        id: NUMBER
        tableNumber: NUMBER
        openedTime: TEXT
        isPaid: BOOLEAN
        items: LIST OF ORDERITEM

    CONSTRUCTOR (i: NUMBER, tNum: NUMBER, oTime: TEXT)
        SET id TO i
        SET tableNumber TO tNum
        SET openedTime TO oTime
        SET isPaid TO FALSE
    END CONSTRUCTOR

    METHODS
        ADDITEM (newItem: ORDERITEM)
            ADD newItem TO LIST items
        END METHOD
        CALCULATETOTAL ()
            DISPLAY "Adds up the price of every item by do this formula (quantity x dish price)"
        END METHOD
        MARKASPAID ()
            SET isPaid TO TRUE
        END METHOD

END CLASS

CLASS ORDERITEM

    ATTRIBUTES
        id: NUMBER
        quantity: NUMBER
        specialRequest: TEXT
        
    CONSTRUCTOR (oi: NUMBER, qty: NUMBER, request: TEXT)
        SET id TO oi
        SET quantity TO qty
        SET specialRequest TO request
    END CONSTRUCTOR

    METHODS
        LINETOTAL ()
            DISPLAY "Multiplies its quantity by the dish price"
        END METHOD

        CHANGEQUANTITY (newQty: NUMBER)
            SET quantity TO newQty
        END METHOD

END CLASS

CLASS DISH

    ATTRIBUTES
        id: NUMBER
        name: TEXT
        price: NUMBER
        prepTimeMinutes: NUMBER
        isAvailable: BOOLEAN

    CONSTRUCTOR (di: NUMBER, n: TEXT, p: NUMBER, prepTime: NUMBER)
        SET id TO di
        SET name TO n
        SET price TO p
        SET prepTimeMinutes TO prepTime
        SET isAvailable TO TRUE
    END CONSTRUCTOR

    METHODS
        MARKUNAVAILABLE()
            SET isAvailable TO FALSE
        END METHOD
 END CLASS

Pseudocode:
START
	BEGIN
		CREATE OBJECT dish1 OF DISH(301, "Chicken Biryani", 3.500, 20)
    		CREATE OBJECT dish2 OF DISH(302, "Hummus & Khaboos", 1.200, 5)
    		CREATE OBJECT dish3 OF DISH(303, "Mango Juice", 0.800, 3)
	
		CREATE OBJECT order1 OF ORDER(1001, 5, "19:30")

		CREATE OBJECT item1 OF ORDERITEM(5001, 2, "Extra spicy", dish1)
		SET order1.ADDITEM(item2)

		SET item2.CHANGEQUANTITY(2)
	END
STOP

CLASS FLIGHT
    ATTRIBUTES
        id: NUMBER
        flightNumber: TEXT
        origin: TEXT

        departureTime: TEXT
        aircraft: AIRCRAFT
        bookings: LIST OF BOOKING

    CONSTRUCTOR (fi: NUMBER, fNum: TEXT, orig: TEXT, dest: TEXT, dTime: TEXT, acObj: AIRCRAFT)
        SET id TO fi
        SET flightNumber TO fNum
        SET origin TO orig
        SET destination TO dest
        SET departureTime TO dTime
        SET aircraft TO acObj
    END CONSTRUCTOR

    METHODS
        ADDBOOKING (newBooking: BOOKING)
            ADD newBooking TO LIST bookings
        END METHOD

        COUNTFREESEATS ()
            DISPLAY "Compare the aircraft's seats with the bookings to see how many are left"
        END METHOD

Blueprint:

CLASS PRODUCT
    ATTRIBUTES
        id: NUMBER
        productName: TEXT
        productCode: TEXT
        productType: TEXT
	isAvailble: BOOLEAN

    CONSTRUCTOR (pName: TEXT, pCode: TEXT, pType: TEXT)
        SET productName TO pName
        SET productCode TO pCode
        SET productType TO pType
    END CONSTRUCTOR

    METHODS
        DISPLAYPRODUCT ()
            DISPLAY "These are the avaible products" + productName
        END METHOD

        ADDPRODUCT ()
            DISPLAY "Enter a new product"
        END METHOD

        CHECKAVAILABILITY ()
            SET isAvailable TO TRUE
        END METHOD

        DELETEPRODUCT ()
            SET isAvailable TO FALSE
        END METHOD
END CLASS

CLASS PURCHASE
    ATTRIBUTES
        custId: NUMBER
        purchaseDate: TEXT
        totalAmount: TEXT
        items: LIST OF LINEITEM

    CONSTRUCTOR (pDate: TEXT, total: TEXT)
        SET purchaseDate TO pDate
        SET totalAmount TO total
    END CONSTRUCTOR

    METHODS
        DISPLAYPURCHASES()
            DISPLAY "Display all purchased items in this transaction"
        END METHOD

        CHECKDATE()
            RETURN purchaseDate
        END METHOD

        CALCULATETOTAL()
            DISPLAY "Adds up the price of every item purchased"
        END METHOD

END CLASS

CLASS LINEITEM
    ATTRIBUTES
        custId: NUMBER
        lineItemId: TEXT
        lineItemName: TEXT
        Quantity: NUMBER
        unitPrice: DOUBLE

    CONSTRUCTOR (lId: TEXT, lName: TEXT, qty: NUMBER, price: DOUBLE)
        SET lineItemId TO lId
        SET lineItemName TO lName
        SET Quantity TO qty
        SET unitPrice TO price
    END CONSTRUCTOR

    METHODS
        DISPLAYLINEITEM()
            DISPLAY lineItemName + " x " + Quantity
        END METHOD

        CALCULATETOTAL ()
            DISPLAY "Calculating subtotal by quantity * unitPrice"
        END METHOD

END CLASS

    ATTRIBUTES

        DeliveryName: TEXT
        DeliveryDay: TEXT
        DeliveryDate: TEXT
        DeliveryTime: TEXT
        isDelivered: BOOLEAN

        SET DeliveryName TO dName
        SET DeliveryDay TO dDay
        SET DeliveryTime TO dTime

    END CLASS
