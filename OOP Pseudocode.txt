Scenario 1: Hospital

Blueprint:

CLASS HOSPITAL
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		address: TEXT
		patients: LIST OF PATIENT

	CONTRUCTOR (i: NUMBER, n: TEXT, a: TEXT)
		SET id TO i
		SET name TO n
		SET address TO a
	END CONTUCTOR

	METHODS
		ADDPATIENT (client: PATIENT)
		END METHOD

		DISCHARGEPATIENT (client: PATIENT, dischargeDate: TEXT)
		END METHOD
END CLASS


CLASS PATIENT
	ATTRIBUTES
		id: NUMBER
		fullName: TEXT
		dateOfBirth: TEXT
		admissionDate:TEXT
		dischargeDate: TEXT
		isAdmitted : BOOLEAN
		room: NUMBER
		procedures: LIST OF PROCEDURE

	CONTRUCTOR (pi : NUMBER, fn: TEXT, dob: TEXT)
		SET id TO pi
		SET fullName TO fn
		SET dateOfBirth TO dob
	END CONSTRUCTOR

	METHODS
		ADDPROCEDURE (pro : PROCEDURE)
			ADD pro TO LIST PROCEDURE
		END METHOD
	
		MOVETOROOM ()
			DISPLAY SIZE OF room
		END METHOD

		TOTALCOSTCARE()
		END METHOD
END CLASS


CLASS PROCEDURE
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		performedDate: TEXT
		admissionDate:TEXT
		doctorName: TEXT
		notes: BOOLEAN
		cost: NUMBER

	METHODS
		DISCRIBEITSELF()
			DISPLAY "Enter a short summary about procedsure"
		END METHOD
	
END CLASS

CLASS ROOM
	ATTRIBUTES
		id: NUMBER
		roomNumber : TEXT
		wardType: TEXT
		floor: NUMBER
		bedCount: NUMBER

	METHODS
		DISPLAYROOM()
			DISPLAY roomNumber
			DISPLAY wardType

CLASS BOOK
	ATTRIBUTES
		id: NUMBER
		title: TEXT
		author: TEXT
		category: TEXT
		compiesAavilable: NUMBER


	CONSTRCUTOR (bi: NUMBER, t: TEXT, aut: TEXT, cat: TEXT, cop:NUMBER)
		SET id TO bi
		SET title TO t
		SET author TO aut
		SET category TO cat
		SET compiesAavilable TO cop
	END CONSTROCUTOR

	METHODS
		TAKEONECOPY()
			SET compiesAavilable to compiesAavilable - 1
		END METHOD

		RETURNONECOPY()
			SET compiesAavilable to compiesAavilable + 1				
		END METHOD
END CLASS

Scenario 3: Restaurant Order

Blueprint:

CLASS ORDER
    ATTRIBUTES
        id: NUMBER
        tableNumber: NUMBER
        openedTime: TEXT
        isPaid: BOOLEAN
        items: LIST OF ORDERITEM

    CONSTRUCTOR (i: NUMBER, tNum: NUMBER, oTime: TEXT)
        SET id TO i
        SET tableNumber TO tNum
        SET openedTime TO oTime
        SET isPaid TO FALSE
    END CONSTRUCTOR

    METHODS
        ADDITEM (newItem: ORDERITEM)
            ADD newItem TO LIST items
        END METHOD
        CALCULATETOTAL ()
            DISPLAY "Adds up the price of every item by do this formula (quantity x dish price)"
        END METHOD
        MARKASPAID ()
            SET isPaid TO TRUE
        END METHOD

END CLASS
----------------------------------------------------------------------------------------------------------------------------
CLASS ORDERITEM

    ATTRIBUTES
        id: NUMBER
        quantity: NUMBER
        specialRequest: TEXT
        
    CONSTRUCTOR (oi: NUMBER, qty: NUMBER, request: TEXT)
        SET id TO oi
        SET quantity TO qty
        SET specialRequest TO request
    END CONSTRUCTOR

    METHODS
        LINETOTAL ()
            DISPLAY "Multiplies its quantity by the dish price"
        END METHOD

        CHANGEQUANTITY (newQty: NUMBER)
            SET quantity TO newQty
        END METHOD

END CLASS

CLASS DISH

    ATTRIBUTES
        id: NUMBER
        name: TEXT
        price: NUMBER
        prepTimeMinutes: NUMBER
        isAvailable: BOOLEAN

    CONSTRUCTOR (di: NUMBER, n: TEXT, p: NUMBER, prepTime: NUMBER)
        SET id TO di
        SET name TO n
        SET price TO p
        SET prepTimeMinutes TO prepTime
        SET isAvailable TO TRUE
    END CONSTRUCTOR

    METHODS
        MARKUNAVAILABLE()
            SET isAvailable TO FALSE
        END METHOD
 END CLASS

Pseudocode:
START
	BEGIN
		CREATE OBJECT dish1 OF DISH(301, "Chicken Biryani", 3.500, 20)

		CREATE OBJECT order1 OF ORDER(1001, 5, "19:30")

		CREATE OBJECT item1 OF ORDERITEM(5001, 2, "Extra spicy", dish1)
		SET order1.ADDITEM(item2)

		SET item2.CHANGEQUANTITY(2)
END
