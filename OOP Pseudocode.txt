Scenario 1: Hospital

Blueprint:

CLASS HOSPITAL
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		address: TEXT
		patients: LIST OF PATIENT

	CONTRUCTOR (i: NUMBER, n: TEXT, a: TEXT)
		SET id TO i
		SET name TO n
		SET address TO a
	END CONTUCTOR

	METHODS
		ADDPATIENT (client: PATIENT)
		END METHOD

		DISCHARGEPATIENT (client: PATIENT, dischargeDate: TEXT)
		END METHOD
END CLASS


CLASS PATIENT
	ATTRIBUTES
		id: NUMBER
		fullName: TEXT
		dateOfBirth: TEXT
		admissionDate:TEXT
		dischargeDate: TEXT
		isAdmitted : BOOLEAN
		room: NUMBER
		procedures: LIST OF PROCEDURE

	CONTRUCTOR (pi : NUMBER, fn: TEXT, dob: TEXT)
		SET id TO pi
		SET fullName TO fn
		SET dateOfBirth TO dob
	END CONSTRUCTOR

	METHODS
		ADDPROCEDURE (pro : PROCEDURE)
			ADD pro TO LIST PROCEDURE
		END METHOD
	
		MOVETOROOM ()
			DISPLAY SIZE OF room
		END METHOD

		TOTALCOSTCARE()
		END METHOD
END CLASS


CLASS PROCEDURE
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		performedDate: TEXT
		admissionDate:TEXT
		doctorName: TEXT
		notes: BOOLEAN
		cost: NUMBER

	METHODS
		DISCRIBEITSELF()
			DISPLAY "Enter a short summary about procedsure"
		END METHOD
	
END CLASS

CLASS ROOM
	ATTRIBUTES
		id: NUMBER
		roomNumber : TEXT
		wardType: TEXT
		floor: NUMBER
		bedCount: NUMBER

	METHODS
		DISPLAYROOM()
			DISPLAY roomNumber
			DISPLAY wardType
		END METHOD
END CLASS

Scenario 2: Library

Blueprint:

CLASS LIBRARY
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		branchAddress: TEXT
		members: LIST OF MEMBER
		books: LIST OF BOOK

	CONSTRCUTOR (i: NUMBER, n: TEXT, a: TEXT)
		SET id TO i
		SET name TO n
		SET branchAddress TO a
	END CONSTROCUTOR

	METHODS
		RESGISGTERMEMBER (newMember: MEMBER)
			ADD newMember TO LIST members
		END METHOD

		ADDBOOK (newBook: BOOK)
			ADD newBook TO LIST books
		END METHOD
	

		FINDMEMBER ()
			DISPLAY "Search for the member in library"
		END METHOD
END CLASS

CLASS MEMBER
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		joinDate: TEXT
		isActive: BOOLEAN
		loans: LIST OF LOAN

	CONSTRCUTOR (i: NUMBER, n: TEXT, jd: TEXT)
		SET id TO i
		SET name TO n
		SET joinDate TO jd
		SET isActive TO TRUE
	END CONSTROCUTOR

	METHODS
		BORROWBOOK()
		END METHOD

		CONUTACTIVELOANS()
			DISPLAY "Counting active loans"
		END METHOD
	
END CLASS

CLASS LOAN
	ATTRIBUTES
		id: NUMBER
		loanDate: TEXT
		dueDate: TEXT
		returnDate: TEXT
		isRetruned: BOOLEAN

	CONSTRCUTOR (li: NUMBER, ldate: TEXT, ddate: TEXT)
		SET id TO li
		SET loanDate TO ldate
		SET dueDate TO ddate
		SET isReturned TO FLASE
	END CONSTROCUTOR

	METHODS
		MARKRETURNED(rDate: TEXT)
			SET isReturned TO TRUE
			SET returnDate TO rDate
		END METHOD

		CHECKOVERDUE()
			DISPLAY "Checking if overdue date" - today
		END METHOD
	
END CLASS

CLASS BOOK
	ATTRIBUTES
		id: NUMBER
		title: TEXT
		author: TEXT
		category: TEXT
		compiesAavilable: NUMBER


	CONSTRCUTOR (bi: NUMBER, t: TEXT, aut: TEXT, cat: TEXT, cop:NUMBER)
		SET id TO bi
		SET title TO t
		SET author TO aut
		SET category TO cat
		SET compiesAavilable TO cop
	END CONSTROCUTOR

	METHODS
		TAKEONECOPY()
			SET compiesAavilable to compiesAavilable - 1
		END METHOD

		RETURNONECOPY()
			SET compiesAavilable to compiesAavilable + 1				
		END METHOD
END CLASS
