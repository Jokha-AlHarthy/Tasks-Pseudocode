Scenario 1: Hospital

Blueprint:

CLASS HOSPITAL
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		address: TEXT
		patients: LIST OF PATIENT

	CONTRUCTOR (i: NUMBER, n: TEXT, a: TEXT)
		SET id TO i
		SET name TO n
		SET address TO a
	END CONTUCTOR

	METHODS
		ADDPATIENT (client: PATIENT)
		END METHOD

		DISCHARGEPATIENT (client: PATIENT, dischargeDate: TEXT)
		END METHOD
END CLASS


CLASS PATIENT
	ATTRIBUTES
		id: NUMBER
		fullName: TEXT
		dateOfBirth: TEXT
		admissionDate:TEXT
		dischargeDate: TEXT
		isAdmitted : BOOLEAN
		room: NUMBER
		procedures: LIST OF PROCEDURE

	CONTRUCTOR (pi : NUMBER, fn: TEXT, dob: TEXT)
		SET id TO pi
		SET fullName TO fn
		SET dateOfBirth TO dob
	END CONSTRUCTOR

	METHODS
		ADDPROCEDURE (pro : PROCEDURE)
			ADD pro TO LIST PROCEDURE
		END METHOD
	
		MOVETOROOM ()
			DISPLAY SIZE OF room
		END METHOD

		TOTALCOSTCARE()
		END METHOD
END CLASS


CLASS PROCEDURE
	ATTRIBUTES
		id: NUMBER
		name: TEXT
		performedDate: TEXT
		admissionDate:TEXT
		doctorName: TEXT
		notes: BOOLEAN
		cost: NUMBER

	METHODS
		DISCRIBEITSELF()
			DISPLAY "Enter a short summary about procedsure"
		END METHOD
	
END CLASS

CLASS ROOM
	ATTRIBUTES
		id: NUMBER
		roomNumber : TEXT
		wardType: TEXT
		floor: NUMBER
		bedCount: NUMBER

	METHODS
		DISPLAYROOM()
			DISPLAY roomNumber
			DISPLAY wardType
		END METHOD
END CLASS
